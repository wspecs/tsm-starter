## Usage

```js
// cat ~/.tsm-starter.json
// {
//   "author": {
//     "name": "Author Name",
//     "email": "author@email.com",
//     "url": "author.url.co"
//   },
//   "repository": "wspecs"
// }

const starter = require('tsm-starer');
starter.generate({
  name: 'new-module',
  config: '~/.tsm-starter.json',
  destination: '~/dest'
});

// Or as binary
tsm-starer new-module --config=~/.tsm-starter.json destination=~/dest

```
